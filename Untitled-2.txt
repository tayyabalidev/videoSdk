"use client";

import { useState, useMemo } from "react";
import Link from "next/link";
import { Sidebar } from "@/components/sidebar";
import { StatCard } from "@/components/strategies/stat-card";
import { Filters } from "@/components/strategies/filters";
import {
  StrategiesTable,
  type Strategy,
} from "@/components/strategies/strategies-table";
import { Pagination } from "@/components/strategies/pagination";
import { DeleteModal } from "@/components/strategies/delete-modal";
import { mockStrategies, strategyStats } from "@/lib/mock-data";

function CompareIcon({ className }: { className?: string }) {
  return (
    <svg
      className={className}
      width="16"
      height="16"
      viewBox="0 0 24 24"
      fill="none"
      stroke="currentColor"
      strokeWidth="2"
      strokeLinecap="round"
      strokeLinejoin="round"
    >
      <path d="M16 3h5v5" />
      <path d="M8 3H3v5" />
      <path d="M21 3l-7 7" />
      <path d="M3 3l7 7" />
      <path d="M16 21h5v-5" />
      <path d="M8 21H3v-5" />
      <path d="M21 21l-7-7" />
      <path d="M3 21l7-7" />
    </svg>
  );
}

function PlusIcon({ className }: { className?: string }) {
  return (
    <svg
      className={className}
      width="16"
      height="16"
      viewBox="0 0 24 24"
      fill="none"
      stroke="currentColor"
      strokeWidth="2"
      strokeLinecap="round"
      strokeLinejoin="round"
    >
      <path d="M5 12h14" />
      <path d="M12 5v14" />
    </svg>
  );
}

export default function StrategiesPage() {
  // Filter state
  const [searchQuery, setSearchQuery] = useState("");
  const [statusFilter, setStatusFilter] = useState("");
  const [periodFilter, setPeriodFilter] = useState("");
  const [classFilter, setClassFilter] = useState("");

  // Selection state
  const [selectedIds, setSelectedIds] = useState<string[]>([]);

  // Pagination state
  const [currentPage, setCurrentPage] = useState(1);
  const [itemsPerPage, setItemsPerPage] = useState(10);

  // Delete modal state
  const [deleteModalOpen, setDeleteModalOpen] = useState(false);
  const [strategyToDelete, setStrategyToDelete] = useState<Strategy | null>(
    null
  );

  // Strategies state
  const [strategies, setStrategies] = useState<Strategy[]>(mockStrategies);

  // Filter strategies
  const filteredStrategies = useMemo(() => {
    return strategies.filter((strategy) => {
      const matchesSearch =
        !searchQuery ||
        strategy.name.toLowerCase().includes(searchQuery.toLowerCase());
      const matchesStatus =
        !statusFilter || strategy.status === statusFilter;
      const matchesPeriod =
        !periodFilter || strategy.testPeriod === periodFilter;
      const matchesClass =
        !classFilter || strategy.class === classFilter;

      return matchesSearch && matchesStatus && matchesPeriod && matchesClass;
    });
  }, [strategies, searchQuery, statusFilter, periodFilter, classFilter]);

  // Paginate
  const totalPages = Math.ceil(filteredStrategies.length / itemsPerPage);
  const paginatedStrategies = filteredStrategies.slice(
    (currentPage - 1) * itemsPerPage,
    currentPage * itemsPerPage
  );

  // Handle delete
  const handleDeleteClick = (strategy: Strategy) => {
    setStrategyToDelete(strategy);
    setDeleteModalOpen(true);
  };

  const handleDeleteConfirm = () => {
    if (strategyToDelete) {
      setStrategies((prev) =>
        prev.filter((s) => s.id !== strategyToDelete.id)
      );
      setSelectedIds((prev) =>
        prev.filter((id) => id !== strategyToDelete.id)
      );
    }
    setDeleteModalOpen(false);
    setStrategyToDelete(null);
  };

  const handlePageChange = (page: number) => {
    setCurrentPage(page);
    setSelectedIds([]);
  };

  const handleItemsPerPageChange = (count: number) => {
    setItemsPerPage(count);
    setCurrentPage(1);
    setSelectedIds([]);
  };

  return (
    <div className="min-h-screen bg-background">
      <Sidebar />

      <main className="pl-56">
        {/* Top bar */}
        <header className="flex h-16 items-center justify-between border-b border-border bg-card px-6">
          <div className="flex items-center gap-2">
            <h2 className="text-sm font-medium text-foreground">Strategies</h2>
          </div>
          <div className="flex items-center gap-3">
            <button className="flex h-8 w-8 items-center justify-center rounded-lg text-muted-foreground hover:bg-muted hover:text-foreground">
              <svg
                width="20"
                height="20"
                viewBox="0 0 24 24"
                fill="none"
                stroke="currentColor"
                strokeWidth="2"
                strokeLinecap="round"
                strokeLinejoin="round"
              >
                <path d="M12 3a6 6 0 0 0 9 9 9 9 0 1 1-9-9Z" />
              </svg>
            </button>
            <button className="flex h-8 w-8 items-center justify-center rounded-lg text-muted-foreground hover:bg-muted hover:text-foreground">
              <svg
                width="20"
                height="20"
                viewBox="0 0 24 24"
                fill="none"
                stroke="currentColor"
                strokeWidth="2"
                strokeLinecap="round"
                strokeLinejoin="round"
              >
                <path d="M6 8a6 6 0 0 1 12 0c0 7 3 9 3 9H3s3-2 3-9" />
                <path d="M10.3 21a1.94 1.94 0 0 0 3.4 0" />
              </svg>
            </button>
            <div className="h-8 w-8 rounded-full bg-primary" />
          </div>
        </header>

        {/* Content */}
        <div className="p-6">
          {/* Page header */}
          <div className="mb-6 flex flex-col gap-4 sm:flex-row sm:items-start sm:justify-between">
            <div>
              <h1 className="text-2xl font-semibold tracking-tight text-foreground">
                Strategies Library
              </h1>
              <p className="mt-1 text-sm text-muted-foreground">
                Manage and analyze your strategy backtests
              </p>
            </div>
            <div className="flex items-center gap-3">
              <button className="inline-flex items-center gap-2 rounded-lg border border-border bg-card px-4 py-2 text-sm font-medium text-foreground transition-colors hover:bg-muted">
                <CompareIcon />
                Compare
              </button>
              <Link
                href="/strategies/create"
                className="inline-flex items-center gap-2 rounded-lg bg-primary px-4 py-2 text-sm font-medium text-primary-foreground transition-colors hover:bg-primary/90"
              >
                <PlusIcon />
                Create Backtest
              </Link>
            </div>
          </div>

          {/* Stats */}
          <div className="mb-6 grid grid-cols-1 gap-4 sm:grid-cols-2 lg:grid-cols-4">
            <StatCard
              label="Active backtests"
              value={strategyStats.activeStrategies.toString()}
              subtitle="Across all strategies"
            />
            <StatCard
              label="Average return"
              value={strategyStats.averageReturn}
              subtitle="For completed backtests"
            />
            <StatCard
              label="Best win rate"
              value={strategyStats.bestWinRate}
              subtitle="Top performing strategy"
            />
            <StatCard
              label="Total backtests"
              value={strategyStats.totalStrategies.toString()}
              subtitle="All time backtests"
            />
          </div>

          {/* Filters */}
          <div className="mb-6">
            <Filters
              searchQuery={searchQuery}
              onSearchChange={setSearchQuery}
              statusFilter={statusFilter}
              onStatusChange={setStatusFilter}
              periodFilter={periodFilter}
              onPeriodChange={setPeriodFilter}
              classFilter={classFilter}
              onClassChange={setClassFilter}
            />
          </div>

          {/* Results count */}
          <p className="mb-4 text-sm text-muted-foreground">
            {filteredStrategies.length} strategies found
          </p>

          {/* Table */}
          <div className="mb-6">
            <StrategiesTable
              strategies={paginatedStrategies}
              selectedIds={selectedIds}
              onSelectionChange={setSelectedIds}
              onDelete={handleDeleteClick}
            />
          </div>

          {/* Pagination */}
          {totalPages > 0 && (
            <Pagination
              currentPage={currentPage}
              totalPages={totalPages}
              totalItems={filteredStrategies.length}
              itemsPerPage={itemsPerPage}
              onPageChange={handlePageChange}
              onItemsPerPageChange={handleItemsPerPageChange}
            />
          )}
        </div>
      </main>

      {/* Delete Modal */}
      <DeleteModal
        isOpen={deleteModalOpen}
        strategyName={strategyToDelete?.name ?? ""}
        onClose={() => {
          setDeleteModalOpen(false);
          setStrategyToDelete(null);
        }}
        onConfirm={handleDeleteConfirm}
      />
    </div>
  );
}
